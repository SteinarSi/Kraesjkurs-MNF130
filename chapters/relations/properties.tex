\subsection{Egenskaper}
\begin{frame}[fragile]{Egenskaper på relasjoner}
Gitt en relasjon $R \subseteq A \times A$ på $A$ har vi 5 viktige begreper for å beskrive den:
    \begin{itemize}
        \item Refleksiv: $\forall a : (a, a) \in R$
        \pause
        \item Symmetrisk: $\forall a, b : (a, b) \in R \leftrightarrow (b, a) \in R$
        \pause
        \item Antisymmetrisk: $\forall a, b : (a, b), (b, a) \in R \rightarrow a = b$
        \pause
        \item Transitiv: $\forall a, b, c : (a, b), (b, c) \in R \rightarrow (a, c) \in R$\\
        \pause
        \item Om en relasjon er refleksiv, symmetrisk og transitiv, kaller vi det en \emph{ekvivalensrelasjon}.
    \end{itemize}   
\end{frame}

\begin{frame}[fragile]{Refleksive relasjoner}
En relasjon $R$ er \emph{refleksiv} hvis $\forall a : ((a, a) \in R)$.\\

            c \arrow[rruu]          &  & d \arrow[loop, distance=2em, in=305, out=235]
            \end{tikzcd}
        }  
        \end{column}

\begin{frame}[fragile]{Symmetriske relasjoner}
    En relasjon $R$ er \emph{symmetrisk} hvis $\forall a, b : (a, b) \in R \rightarrow (b, a) \in R$.\\
    \begin{columns}
        \begin{column}{0.3\textwidth}
            Ikke symmetrisk:\\
            \adjustbox{scale=0.8}{

                    c \arrow[ru] \arrow[u, bend right] & d \arrow[loop, distance=2em, in=35, out=325]

\begin{frame}[fragile]{Transitive relasjoner}
    En relasjon $R$ er \emph{transitiv} om $\forall a, b, c : (a, b), (b, c) \in R \rightarrow (a, c) \in R$.\\
    \begin{columns}
        \begin{column}{0.3\textwidth}
            Ikke transitiv:
             \begin{tikzcd}
            a \arrow[r] \arrow[d] & b \arrow[d] \\
            d                     & c          
            \end{tikzcd}
        \end{column}
        \pause
        \begin{column}{0.3\textwidth}
            Transitiv:
            \begin{tikzcd}
            a \arrow[r] \arrow[d] \arrow[rd] & b \arrow[d] \\
            d                                & c          
            \end{tikzcd}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}[fragile]{Ekvivalenserelasjoner}
    En ekvivalenserelasjon $R \subseteq A \times A$ er en \emph{ekvivalenserelasjon} om den er både refleksiv, symmetrisk, og transitiv. Da oppfører den seg mye som $=$ og $\equiv$ gjør.\\
    \begin{columns}
        \begin{column}{0.4\textwidth}
            Ikke en ekvivalenserelasjon:
            \begin{tikzcd}
                a \arrow[d, bend right]            & b                                                      & e \\
                c \arrow[ru] \arrow[u, bend right] & d \arrow[r] \arrow[loop, distance=2em, in=125, out=55] & c
            \end{tikzcd}
        \end{column}
        \pause
        \begin{column}{0.4\textwidth}
            En ekvivalenserelasjon:
            \begin{tikzcd}
                a \arrow[d, bend right] \arrow[loop, distance=2em, in=125, out=55] \arrow[r, bend left]    & b \arrow[loop, distance=2em, in=125, out=55] \arrow[ld, bend right] \arrow[l, bend left] & e \arrow[loop, distance=2em, in=125, out=55]                        \\
                c \arrow[ru, bend right] \arrow[u, bend right] \arrow[loop, distance=2em, in=305, out=235] & d \arrow[r, bend right] \arrow[loop, distance=2em, in=305, out=235]                      & c \arrow[loop, distance=2em, in=305, out=235] \arrow[l, bend right]
            \end{tikzcd}
        \end{column}
    \end{columns}
\end{frame}
